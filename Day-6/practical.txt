# Day 6 - Practical 6: docker pull nginx

# Step 1: Open terminal

# Step 2: Pull the Nginx image
docker pull nginx

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the Nginx container in detached mode
docker run -d -p 8080:80 nginx

# Step 5: Verify the container is running
docker ps

# Step 6: Access Nginx in browser
# http://localhost:8080

# Step 7: Stop the container
docker stop $(docker ps -q --filter ancestor=nginx)
