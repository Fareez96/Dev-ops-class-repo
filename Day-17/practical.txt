# Day 17 - Practical 17: docker pull openjdk

# Step 1: Open terminal

# Step 2: Pull the OpenJDK image
docker pull openjdk

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the OpenJDK container in interactive mode
docker run -it openjdk

# Step 5: Inside the container, check Java version
java -version

# Step 6: Exit the container
exit

# Step 7: Verify the container status
docker ps -a
