# Day 12 - Practical 12: docker pull redis

# Step 1: Open terminal

# Step 2: Pull the Redis image
docker pull redis

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the Redis container
docker run -d -p 6379:6379 redis

# Step 5: Verify the container is running
docker ps

# Step 6: Access Redis CLI inside the container
docker exec -it $(docker ps -q --filter ancestor=redis) redis-cli

# Step 7: Inside Redis, test with ping
ping

# Step 8: Exit Redis
exit
