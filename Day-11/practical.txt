# Day 11 - Practical 11: docker pull mongo

# Step 1: Open terminal

# Step 2: Pull the MongoDB image
docker pull mongo

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the MongoDB container
docker run -d -p 27017:27017 mongo

# Step 5: Verify the container is running
docker ps

# Step 6: Access MongoDB inside the container
docker exec -it $(docker ps -q --filter ancestor=mongo) mongosh

# Step 7: Inside MongoDB, show databases
show dbs

# Step 8: Exit MongoDB
exit
