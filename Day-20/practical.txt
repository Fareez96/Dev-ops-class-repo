# Day 20 - Practical 20: docker pull golang

# Step 1: Open terminal

# Step 2: Pull the Golang image
docker pull golang

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the Golang container in interactive mode
docker run -it golang

# Step 5: Inside the container, check Go version
go version

# Step 6: Exit the container
exit

# Step 7: Verify the container status
docker ps -a
