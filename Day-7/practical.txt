# Day 7 - Practical 7: docker pull nginx:alpine

# Step 1: Open terminal

# Step 2: Pull the Nginx Alpine image
docker pull nginx:alpine

# Step 3: Verify the image is downloaded
docker images

# Step 4: Run the Nginx Alpine container in detached mode
docker run -d -p 8081:80 nginx:alpine

# Step 5: Verify the container is running
docker ps

# Step 6: Access Nginx Alpine in browser
# http://localhost:8081

# Step 7: Stop the container
docker stop $(docker ps -q --filter ancestor=nginx:alpine)
